def initialize(hash, bot = nil)
  super :Quote, bot
  @id = hash['id']
  @group_id = hash['groupId']
  @sender_id = hash['senderId']
  @target_id = hash['targetId']
  @origin = MessageChain.make(*hash['origin'], sender_id: @sender_id, bot: bot)
end
