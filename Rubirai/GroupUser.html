def initialize(hash, bot = nil)
  raise(RubiraiError, 'not a group user') unless hash.key? 'group'
  super(hash, bot)
  @member_name = hash['memberName']
  @permission = hash['permission']
  @group = Group.new(hash['group'], bot)
end
